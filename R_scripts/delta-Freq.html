#Evaluate delta-freq 

deltaOut2<-vector("list",length(GEVApos))
for (r in 1:length(deltaOut2)){

combineAge = merge(GEVApos[[r]],slimAge[[r]],by="Position")
combineAge<-combineAge[combineAge$frequency<1, ]
#combineAge<-combineAge[combineAge$frequency>0.002, ]

TE<-combineAge[combineAge$type!="SNP", ]
TE<-TE[order(TE$PostMode), ]
TE$ageBin <- as.numeric(cut_number(TE$PostMode,10))
TE_sp<-split(TE,TE$ageBin)

SNPs<-combineAge[combineAge$type=="SNP", ]
snpBINS<-lapply(TE_sp,function(df) SNPs[between(SNPs$PostMode,min(df[ ,"PostMode"]),max(df[ ,"PostMode"])), ])

#downsample
for (i in 1:length(snpBINS)){
	mySNP<-snpBINS[[i]]
	mySNP<-mySNP[sample(nrow(mySNP),nrow(TE_sp[[i]]),replace=F), ]
    ID=names(snpBINS)[i]
    mySNP$ageBin<-ID
	snpBINS[[i]]<-mySNP
}

afMean_snp<-lapply(snpBINS,function(df) mean(df$frequency,na.rm=TRUE))
afMean_TE<-lapply(TE_sp,function(df) mean(df$frequency,na.rm=TRUE))  

ksTEST<-NULL
nTE<-NULL
nSNP<-NULL
for (i in 1:length(snpBINS)){
    ID=names(snpBINS)[i]
    ksTEST[i] = ks.test(snpBINS[[i]]$frequency,TE_sp[[i]]$frequency)$p.value
    nSNP[i]<-nrow(snpBINS[[i]])
    nTE[i]<-nrow(TE_sp[[i]])
}
  
                  
forPlot<-cbind.data.frame(ageBin=names(snpBINS),snpMean=unlist(afMean_snp),TEmean=unlist(afMean_TE),
                          ksTEST,nSNP,nTE)
forPlot$delta<-forPlot$TEmean - forPlot$snpMean #flipping this to match the order in Robert's paper

deltaOut2[[r]]<-forPlot
}
